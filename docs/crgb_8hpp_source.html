154FASTLED_FORCE_INLINE uint8_t CRGB::getLuma( ) const {
155 //Y' = 0.2126 R' + 0.7152 G' + 0.0722 B'
156 // 54 183 18 (!)
157
158 uint8_t luma = scale8_LEAVING_R1_DIRTY( r, 54) + \
159 scale8_LEAVING_R1_DIRTY( g, 183) + \
160 scale8_LEAVING_R1_DIRTY( b, 18);
161 cleanup_R1();
162 return luma;
163}
