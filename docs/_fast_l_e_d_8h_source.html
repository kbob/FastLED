397 template<ESPIChipsets CHIPSET, uint8_t DATA_PIN, uint8_t CLOCK_PIN>
398 struct ClockedChipsetHelper {
399 // Default implementation, will be specialized for supported chipsets
400 static const bool IS_VALID = false;
401 };
402
403 // Macro to define a mapping from the ESPIChipeset enum to the controller class
404 // in it's various template configurations.
405 #define _FL_MAP_CLOCKED_CHIPSET(CHIPSET_ENUM, CONTROLLER_CLASS) \
406 template<uint8_t DATA_PIN, uint8_t CLOCK_PIN> \
407 struct ClockedChipsetHelper<CHIPSET_ENUM, DATA_PIN, CLOCK_PIN> { \
408 static const bool IS_VALID = true; \
409 typedef CONTROLLER_CLASS<DATA_PIN, CLOCK_PIN> ControllerType; \
410 /* Controller type with RGB_ORDER specified */ \
411 template<EOrder RGB_ORDER> \
412 struct CONTROLLER_CLASS_WITH_ORDER { \
413 typedef CONTROLLER_CLASS<DATA_PIN, CLOCK_PIN, RGB_ORDER> ControllerType; \
414 }; \
415 /* Controller type with RGB_ORDER and spi frequency specified */ \
416 template<EOrder RGB_ORDER, uint32_t FREQ> \
417 struct CONTROLLER_CLASS_WITH_ORDER_AND_FREQ { \
418 typedef CONTROLLER_CLASS<DATA_PIN, CLOCK_PIN, RGB_ORDER, FREQ> ControllerType; \
419 }; \
420 };
421
422 // Define specializations for each supported CHIPSET
423 _FL_MAP_CLOCKED_CHIPSET(LPD6803, LPD6803Controller)
424 _FL_MAP_CLOCKED_CHIPSET(LPD8806, LPD8806Controller)
425 _FL_MAP_CLOCKED_CHIPSET(WS2801, WS2801Controller)
426 _FL_MAP_CLOCKED_CHIPSET(WS2803, WS2803Controller)
427 _FL_MAP_CLOCKED_CHIPSET(SM16716, SM16716Controller)
428 _FL_MAP_CLOCKED_CHIPSET(P9813, P9813Controller)
429
430 // Both DOTSTAR and APA102 use the same controller class
431 _FL_MAP_CLOCKED_CHIPSET(DOTSTAR, APA102Controller)
432 _FL_MAP_CLOCKED_CHIPSET(APA102, APA102Controller)
433
434 // Both DOTSTARHD and APA102HD use the same controller class
435 _FL_MAP_CLOCKED_CHIPSET(DOTSTARHD, APA102ControllerHD)
436 _FL_MAP_CLOCKED_CHIPSET(APA102HD, APA102ControllerHD)

443 template<ESPIChipsets CHIPSET, uint8_t DATA_PIN, uint8_t CLOCK_PIN, EOrder RGB_ORDER, uint32_t SPI_DATA_RATE > CLEDController &addLeds(struct CRGB *data, int nLedsOrOffset, int nLedsIfOffset = 0) {
444 // Instantiate the controller using ClockedChipsetHelper
445 typedef ClockedChipsetHelper<CHIPSET, DATA_PIN, CLOCK_PIN> CHIP;
446 typedef typename CHIP::template CONTROLLER_CLASS_WITH_ORDER_AND_FREQ<RGB_ORDER, SPI_DATA_RATE>::ControllerType ControllerTypeWithFreq;
447 static_assert(CHIP::IS_VALID, "Unsupported chipset");
448 static ControllerTypeWithFreq c;
449 return addLeds(&c, data, nLedsOrOffset, nLedsIfOffset);
450 }
